# ============================================
# GOOGLE CLOUD SERVICE ACCOUNT (Required)
# ============================================
# This is needed for photo uploads and blessings/RSVP submissions
# 
# How to get it:
# 1. Go to https://console.cloud.google.com/
# 2. Create a new project (or select existing)
# 3. Enable APIs: Google Drive API and Google Sheets API
# 4. Go to IAM & Admin > Service Accounts
# 5. Create Service Account (or use existing)
# 6. Click on the service account > Keys tab > Add Key > Create new key
# 7. Choose JSON format
# 8. Download the JSON file
# 9. Open the JSON file and copy the ENTIRE contents (including {})
# 10. Paste it below as a single line (remove line breaks)
#
# Example format:
# GOOGLE_SERVICE_ACCOUNT_JSON={"type":"service_account","project_id":"your-project-id","private_key_id":"...","private_key":"-----BEGIN PRIVATE KEY-----\n...\n-----END PRIVATE KEY-----\n","client_email":"...","client_id":"...","auth_uri":"...","token_uri":"...","auth_provider_x509_cert_url":"...","client_x509_cert_url":"..."}
#
GOOGLE_SERVICE_ACCOUNT_JSON=

# ============================================
# GOOGLE OAUTH FOR DRIVE UPLOADS (Required for photo upload)
# ============================================
# Photo uploads use OAuth so files are stored in your personal Google Drive.
# Create an OAuth client in Google Cloud, then run:
#
# npm run oauth:drive-token
#
# The script prints GOOGLE_REFRESH_TOKEN one time. Add these values to .env
# locally and to Netlify environment variables. Do not commit real values.
#
GOOGLE_CLIENT_ID=
GOOGLE_CLIENT_SECRET=
GOOGLE_REFRESH_TOKEN=

# ============================================
# GOOGLE DRIVE FOLDER ID (Required)
# ============================================
# This is where uploaded photos will be stored
# 
# How to get it:
# 1. Go to https://drive.google.com/
# 2. Create a new folder (or use existing) for wedding photos
# 3. Open the folder
# 4. Look at the URL in your browser
# 5. The URL will look like: https://drive.google.com/drive/folders/ABC123XYZ789
# 6. Copy the part after "/folders/" - that's your folder ID
# 7. Share the folder with your service account email (found in the JSON above)
# 8. Give it "Editor" permissions
#
# Example:
# GOOGLE_DRIVE_FOLDER_ID=ABC123XYZ789
#
GOOGLE_DRIVE_FOLDER_ID=

# ============================================
# GOOGLE SHEETS SPREADSHEET ID (Required)
# ============================================
# This is where blessings and RSVP data will be stored
# 
# How to get it:
# 1. Go to https://sheets.google.com/
# 2. Create a new spreadsheet (or use existing)
# 3. Open the spreadsheet
# 4. Look at the URL in your browser
# 5. The URL will look like: https://docs.google.com/spreadsheets/d/ABC123XYZ789/edit
# 6. Copy the part after "/d/" and before "/edit" - that's your spreadsheet ID
# 7. Share the spreadsheet with your service account email (found in the JSON above)
# 8. Give it "Editor" permissions
# 9. (Optional) Add headers in row 1: Name, Email, Message, Timestamp
#
# Example:
# GOOGLE_SPREADSHEET_ID=ABC123XYZ789
#
GOOGLE_SPREADSHEET_ID=

# ============================================
# UPLOAD PAGE CONFIG SHEET (Optional)
# ============================================
# The upload page can read text and colors from a tab in GOOGLE_SPREADSHEET_ID.
# Create a tab named UploadConfig with columns:
# A: key
# B: value
#
# Example keys:
# albumLabel, eventName, instructions, cameraButton, anotherPhotoButton,
# uploadingMessage, successMessage, errorMessage, selectFileButton,
# backgroundStartColor, backgroundMiddleColor, backgroundEndColor,
# primaryColor, primaryDarkColor, textColor, mutedTextColor
#
GOOGLE_UPLOAD_CONFIG_SHEET_NAME=UploadConfig

# ============================================
# RSVP API URL (Optional)
# ============================================
# Only needed if you want to use Google Apps Script for RSVP form submissions
# If you leave this empty, RSVP will still work but won't submit to Google Sheets
# 
# How to get it (if using Google Apps Script):
# 1. Create a Google Apps Script project
# 2. Write a script to handle RSVP submissions
# 3. Deploy as a web app
# 4. Copy the web app URL
# 5. Paste it below
#
# Example:
# REACT_APP_RSVP_API_URL=https://script.google.com/macros/s/YOUR_SCRIPT_ID/exec
#
# Leave empty if not using:
REACT_APP_RSVP_API_URL=
